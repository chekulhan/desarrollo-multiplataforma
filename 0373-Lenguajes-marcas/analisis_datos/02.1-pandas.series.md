```python
import pandas as pd

s = pd.Series([1, 5, 7, 2, 8])
s = pd.Series([1, 5, 7.5, 2, 8])  
s = pd.Series([1, 'hello', 3.5])
print(type(s))
print(s.dtype)
s.describe()
```

# access elements and sum, etc

```python
s = pd.Series([1, 5, 7, 2, 8])
print(s.iloc[0])
print(s.iloc[0] + s.iloc[2])
s.iloc[2:4]
s.sum()
s.count()


s = pd.Series([1, 5, 7], index=['a', 'b', 'c'])
s = pd.Series([1, 2.5,0, -4], index= ['a', 'b', 'c', 'd'], name='Values')
print(s['a'])
print(s.loc['b'])

```
